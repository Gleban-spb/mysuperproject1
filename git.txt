Если нужна работа через git.com
Создать ноый аккаунт на git.com

git pull #Загрузил файлы из интернета
history >hist_20200927.txt #Создал новый файл
git status #Проверил изменения
git add . #Добавил изменения 
git commit -m "Добавил историю 20220927"  #Создал новый комит с комментом
git push origin #Выгрузил изменения в интернет
git log --pretty=oneline #проверил лог укороченный вариант
